<script>
    $(function () {
        var send = $("#send");
        var clear = $("#clear");
        var wall = $("#wall");
        var i = 0;
        var spanArray = $("#wall").children();
        var speed = 5;

        send.click(function () {
            var danmu = $("#input1").val();
            var id = "span" + i;
            //console.log(id);
            var txt=$("<span id=" + "\'"+ id +"\' "+ "></span>").text(danmu);
            //console.log(txt);
            var top = parseInt(Math.random() * 500) - 20;
            top = top < 0 ? 0 : top;
            wall.append(txt);
            spanArray = $("#wall").children();
            //$('#input1').val(" ");
            //console.log(txt.value);
            var color = new Color();
            //console.log(color.color)
            //wall.css("background-color",color.color);
            //var color = Color();
            //console.log(wall.css("background-color"));
            $("#" + id).css(
                {   "position": "absolute",
                    "top": top + "px",
                    "color": color.color,
                    "left": "1350px",
                    "whiteSpace": "nowrap"

                });
            var num = (Math.random() * 10) + 1;
            txt.attr('spd', num);
            i++;
            $('#input1').val(" ");

        });


        setInterval(function () {
            //spanArray[0].style.left = spanArray[0].style.left - 10 + "px";
            if(spanArray.length !== 0){
                spanArray = $("#wall").children();

                for (var i=0; i<spanArray.length; i++){
                    spanArray[i].style.left =
                        parseInt(spanArray[i].style.left) - speed + "px";
                }

            }
        }, 50);

        for(var i=0; i<spanArray.length; i++){
            var left = spanArray[i].style.left;
            console.log(left);
            if (left < 0){
                var id = "span" + i;
                console.log(id);
                $("#" + id).remove();
                //wall.remove(spanArray[i]);
            }
        }


        // setInterval(move, 200);
        // function move() {
        //     var spanArray = $('#wall').children();

        //         spanArray[i].style.left =
        //             parseInt(spanArray[i].style.left) - spanArray[i].speed + 'px';
        //     }
        // }

        clear.click(function () {
            $("#wall").text("");
            spanArray = [];
        });

        function Color(){
            this.r = Math.floor(Math.random()*255);
            this.g = Math.floor(Math.random()*255);
            this.b = Math.floor(Math.random()*255);
            this.color = 'rgba('+ this.r +','+ this.g +','+ this.b +',0.8)';
        }

    })
</script>
