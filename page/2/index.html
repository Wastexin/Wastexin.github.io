import requests
import re
import json
from requests.exceptions import RequestException
import time

def get_one_page(url):
    #使用try...except 获取异常
    try:
        headers = {
            'User-Agent':'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWe'
                         'bKit/537.36 (KHTML, like Gecko) Chrome/69.0.3497.100 Safari/537.36)'
        }
        response = requests.get(url, headers=headers)
        if response.status_code == 200:
            return response.text
        return None
    except RequestException:
        return None

def paese_one_page(html):
    pattern = re.compile(
        '<dd>.*?board-index.*?>(.*?)</i>.*?data-src="(.*?)".*?name.*?a.*?>(.*?)'
        '</a>.*?star.*?>(.*?)</p>.*?releasetime.*?>(.*?)</p>.*?integer.*?>(.*?)'
        '</i>.*?fraction.*?>(.*?)</i>.*?</dd>',re.S)

    items = re.findall(pattern,html)  #第二个参数为放进去处理的文本或字符串
    for item in items:
        #在一个函数中，程序执行到yield语句的时候，程序暂停，返回yield后面表达式
        #的值，在下一次调用的时候，从yield语句暂停的地方继续执行，如此循环，直到函数执行完。
        yield {
            'index':item[0],
            'image':item[1],
            'title':item[2].strip(),#删除开头或结尾的字符串
            'actor':item[3].strip()[3:],
            'time':item[4].strip()[5:],
            'score':item[5].strip() + item[6].strip()
        }
        #迭代生成器,相当于依次生成了一个列表
        # dict = {
        #     'index':item[0],
        #     'img':item[1],
        #     'title':item[2],
        #     'actor':item[3].strip()[3:] ,
        #     'time':item[4].strip()[5:],
        #     'score':item[5].strip()+item[6].strip()
        # }


def write_to_file(content):
    with open('result.txt','a',encoding='utf-8') as f:
        print(type(json.dumps(content)))  #dumps()将python对象编码成为json对象
        print(content)
        f.write(json.dumps(content,ensure_ascii=False)+'\n')
        #ensure_ascii = False 保证输出为中文而不是Unicode编码


def main(offset):
    url = 'http://maoyan.com/board/4?offset=' + str(offset)
    html = get_one_page(url)
    for item in  paese_one_page(html):
        write_to_file(item)

二、常用的匹配方法

1.match() ：向它传入要匹配的字符串以及正则表达式，就可以检测这个正则表达式是否匹配表达式

match()方法会尝试从字符串的起始位置匹配正则表达式，如果匹配，就会匹配成功的结果；如果不匹配，就会返回None。 示例代码：

import re

content = "Hello world 123456789 Hello_wolrd I am Wx"
print(len(content))
result = re.match('^Hello\s\w{5}\s\d{9}\s\w{11}',content)
#在上述正则表达式中，可以参照常用的匹配模式
#\s匹配任意空白字符，\w匹配任意字母，{}中的数字表示匹配的数量
print(result)
print(result.span())
#span()方法可以输出匹配的范围

输出结果：

41
<_sre.SRE_Match object; span=(0, 33), match='Hello world 123456789 Hello_wolrd'>
(0, 33)
  • 匹配目标：从字符串中提取一部分内容

可以用（）括号将想提取的字符串括号起来。()实际上标记了一个子表达式的开始和结束的位置，被标记的每个子表达式会依次对应一个分组，调用group()方法传入分组的检索即可获取提取的结果。 代码实例：

import re

content = "Hello 123456789 world Hello_wolrd I am Wx"
print(len(content))
result = re.match('^Hello\s(\d+)\sworld',content)
#在上述正则表达式中，可以参照常用的匹配模式
#\s匹配任意空白字符，\w匹配任意字母，{}中的数字表示匹配的数量
print(result)
print(result.group())
print(result.group(0))
print(result.group(1))

输出结果：

41
<_sre.SRE_Match object; span=(0, 21), match='Hello 123456789 world'>
Hello 123456789 world
Hello 123456789 world
123456789

可以看到：result.group()和result.group(0)的效果是等价的。 group()方法的第一个分组（下标为0）是匹配出来的整个字符串，后面用（）括起来的子表达式依次标号有自己的分组号。、

  • 通用匹配：在上述的正则表达式中，出现空白字符就用\s匹配，出现数字用\d匹配，工作量很大。

通用匹配使用的是.（点星）匹配：其中.（点）可以匹配任意字符（除了换行符），（星）代表匹配前前面字符的无限次。 示例代码：

import re

content = "Hello 123456789 world Hello_wolrd I am Wx"
print(len(content))
result = re.match('^Hello.*Wx$',content)
#在中间部分我们全部用.*来代替，并在结尾加上一个结尾字符串$
print(result)
print(result.group())

输出结果：

41
<_sre.SRE_Match object; span=(0, 41), match='Hello 123456789 world Hello_wolrd I am Wx'>
Hello 123456789 world Hello_wolrd I am Wx
  • 贪婪与非贪婪：

.为贪婪匹配：匹配尽量多的字符。 .?为非贪婪匹配：匹配尽量少的字符。

  • 修饰符：可选一些标志修饰符来控制匹配的模式

    conten = “python”
    result = re.match(“^py.*?”,content,re.S)

  • 转义匹配： 正则匹配模式的特殊字符时，在前面加反斜线转义一下 。

2.search():在匹配时会扫描整个字符串，然后返回第一个成功匹配的结果。

3.findall()：搜索整个字符串，然后返回正则表达式匹配的所有内容。返回结果类型为列表 .所以只是获取第一个内容就可以使用search(),提取多个内容则可使用findall()。 4.sub():字符串替换。re.sub(‘\d+’,’ ‘,content):第一个参数为需替换的正则表达式匹配的字符串，第二个参数为替换的字符串，第三个参数为原字符串。

②在multiprocessing模块中提供Pool类来代表进程池对象

#!/usr/bin/python
# -*- coding: utf-8 -*-
'''
@author: WeiXin
'''
# 使用multiprocessing模块创建多进程
import os , time ,random
from multiprocessing import Pool


# 在multiprocessing中提供了一个Process来描述一个进程对象
# 创建子进程时只需要传入一个需要执行的函数和该函数的参数，即可完成对Process类的实例化

# 先定义一个进程运行的函数
def run_proc(name):
    print("I am task %s (pid = %s)" %(name, os.getpid()))
    time.sleep(random.random() * 5)
    print("Task %s end." %name)


if __name__ == '__main__':
    print("Current process pid is %s." %os.getpid())
    p = Pool(processes=4)                #Process默认是CPU的核数，大小受限于CPU
    for i in range(8):
        p.apply_async(run_proc,args=(i,))
    print("waiting for all subprocesses done...")
    p.close()      #Pool对象调用join()方法会等待所有子程序进行完毕
    p.join()       #调用join()之前必须先调用close(),调用close()之
    print("All subprocesses done.")#后就没办法往Pool里添加process了

#在这里我们使用了apply_async()这是异步非阻塞的，意思就是不用等待当前进程执行完毕，随时根据系统调度来进行进程切换。
#还有一种方法apply()是阻塞的,意思就是等待当前子进程执行完毕后，在执行下一个进程。

------------------------------------------------------------ 以下参考博主：xlengji 来自CSDN 原文：https://blog.csdn.net/xlengji/article/details/81084059?utm_source=copy apply是阻塞式的。 首先主进程开始运行，碰到子进程，操作系统切换到子进程，等待子进程运行结束后，在切换到另外一个子进程，直到所有子进程运行完毕。然后在切换到主进程，运行剩余的部分。 apply_async是异步非阻塞式的。 首先主进程开始运行，碰到子进程后，主进程说：让我先运行个够，等到操作系统进行进程切换的时候，在交给子进程运行。以为我们的程序太短，然而还没等到操作系统进行进程切换，主进程就运行完毕了。 想要子进程执行，就告诉主进程：你等着所有子进程执行完毕后，在运行剩余部分。 Python官方建议：废弃apply,使用apply_async。 ————————————————————–
